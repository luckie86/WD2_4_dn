CREATE TABLE JobType (
	JobTypeID INTEGER PRIMARY KEY AUTOINCREMENT,
	Name TEXT
);

CREATE TABLE Region (
	RegionID INTEGER PRIMARY KEY AUTOINCREMENT,
	Name TEXT
);

CREATE TABLE Company (
	CompanyID INTEGER PRIMARY KEY AUTOINCREMENT,
	Name TEXT,
	RegionID INTEGER
);

CREATE TABLE User (
	UserID INTEGER PRIMARY KEY AUTOINCREMENT,
	Email TEXT,
	UserName TEXT,
	Password TEXT,
	Type TEXT,
	CompanyID INTEGER,
	RegionID INTEGER
);

CREATE TABLE JobAd (
	JobAdID INTEGER PRIMARY KEY AUTOINCREMENT,
	CompanyID INTEGER,
	PositionName TEXT,
	Description TEXT,
	JobTypeID INTEGER,
	FullTime TEXT
);
